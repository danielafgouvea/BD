CREATE TABLE clientes (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100),
  email VARCHAR(100)
);
CREATE TABLE produtos (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nome VARCHAR(100),
  preco DECIMAL(10,2)
);
CREATE TABLE vendas (
  id INT AUTO_INCREMENT PRIMARY KEY,
  id_cliente INT,
  id_produto INT,
  data_venda DATE,
  quantidade INT);

SELECT
  vendas.id,
  clientes.nome AS cliente,
  produtos.nome AS produto,
  vendas.data_venda,
  vendas.quantidade,
  produtos.preco,
  (vendas.quantidade * produtos.preco) AS total
FROM vendas
  Join clientes ON vendas.id_cliente = clientes.id
  Join produtos ON vendas.id_produto = produtos.id;
